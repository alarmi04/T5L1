#!/bin/bash
read -p "Introduce un valor: " v1
read -p "Introduce un segundo valor: " v2
if [ $v1 -eq $v2 ]
    then 
        echo "Son iguales"
    else 
        if [ $v1 -gt $v2 ]
            then 
                echo $v1 " es mayor que " $v2
            else 
                echo $v2 " es mayor que " $v1
                fi
                fi
