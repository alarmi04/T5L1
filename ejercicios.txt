#!/bin/bash
read -p "Introduce un valor: " v1
read -p "Introduce un segundo valor: " v2
if [ $v1 -eq $v2 ]
    then 
        echo "Son iguales"
    else 
        if [ $v1 -gt $v2 ]
            then 
                echo $v1 " es mayor que " $v2
            else 
                echo $v2 " es mayor que " $v1
                fi
                fi

#!/bin/bash
read -p "Introduce un numero mayor que 0: " num
let resto=num%2
if [ $resto -eq 0 ]
    then
        echo "Es par"
    else 
        echo "Es impar"
fi

#!/bin/bash
read -p "Introduce una nota: " nota
if [ $nota -lt 5 ] && [ $nota -gt 0 ]
then 
echo "Suspendido"
else 
if [ $nota -lt 6 ]
then 
echo "Aprobado"
else 
if [ $nota -lt 7 ]
then
echo "Bien"
else
if [ $nota -gt 6 ] && [ $nota -lt 9 ] 
then
echo "Notable"
else 
echo "Excelente"
fi
fi
fi
fi
